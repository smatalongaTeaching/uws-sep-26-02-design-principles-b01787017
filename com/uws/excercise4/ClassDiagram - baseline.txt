    }

        +fax(Document)
        +copy(Document)
    }

    class SimplePrinter {
        +print(Document)
        +scan(Document)
        +fax(Document)
        +copy(Document)
    }

    Machine <|.. MultiFunctionPrinter
    Machine <|.. SimplePrinter
    MultiFunctionPrinter ..> Document
